function calc() { // ===== INPUT ===== let load = document.getElementById("load").value; let sun = document.getElementById("sun").value; let panel = document.getElementById("panel").value; let price = document.getElementById("price").value; let cost = document.getElementById("cost").value; let voltage = document.getElementById("voltage").value; let backup = document.getElementById("backup").value; let dod = document.getElementById("dod").value / 100; let distance = document.getElementById("distance").value; let stringSize = document.getElementById("stringSize").value; let eff = document.getElementById("eff").value / 100; let systemType = document.getElementById("systemType").value; // ===== SOLAR ===== let systemKW = (load / sun) / eff; let panels = (systemKW * 1000) / panel; let strings = panels / stringSize; // ===== BATTERY ===== let batteryAh = (load * 1000 * backup) / (voltage * dod); let batteryCount = batteryAh / 100; // สมมุติแบต 100Ah // ===== INVERTER ===== let inverterKW = systemKW * 1.2; let acCurrent = (inverterKW * 1000) / 230; // ===== CABLE ===== let vdropDC = voltage * 0.03; let dcCurrent = (systemKW * 1000) / voltage; let dcCable = (2 * distance * dcCurrent) / (56 * vdropDC); let vdropAC = 230 * 0.03; let acCable = (2 * distance * acCurrent) / (56 * vdropAC); // ===== FINANCE ===== let yearly = load * 365; let saving = yearly * price; let roi = cost / saving; // ===== RESULT ===== document.getElementById("result").innerHTML = "☀️ ระบบ: " + systemType + "

" + "🔋 Solar
" + "ขนาด: " + systemKW.toFixed(2) + " kW

" + "🔋 Battery

" + "จำนวนแบต (100Ah): " + Math.ceil(batteryCount) + "

" + "⚡ Inverter
" + "ขนาด: " + inverterKW.toFixed(2) + " kW

" + "คืนทุน: " + roi.toFixed(1) + " ปี"; }
